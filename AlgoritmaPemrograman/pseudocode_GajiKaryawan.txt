START
    INPUT Golongan
    INPUT JamLembur

    // Declare Gaji Golongan
    SWITCH Golongan
        CASE 'A' :
            GajiPokok = 5000000
        CASE 'B' :
            GajiPokok = 6500000
        CASE 'C' :
            GajiPokok = 9500000
        DEFAULT :
            GajiPokok = 0
    SWITCH END

    // Perhitungan Jam Lembur
    IF JamLembur = 1 THEN
        GajiLembur = GajiPokok * 0.30
    ELSE IF JamLembur = 2 THEN
        GajiLembur = GajiPokok * 0.32
    ELSE IF JamLembur = 3 THEN
        GajiLembur = GajiPokok * 0.34
    ELSE IF JamLembur = 4 THEN
        GajiLembur = GajiPokok * 0.36
    ELSE IF JamLembur = 5 THEN
        GajiLembur = GajiPokok * 0.38
    ELSE
        GajiLembur = 0
    IF END

    // TOTAL GAJI KARYAWAN
    GajiPokok + GajiLembur = TotalGajiKaryawan

    // Output
    OUTPUT TotalGajiKaryawan

END
